//sw为开关的输入，seg为到不带译码的数码管，auto为带译码的数码管
module decoder(
    input wire [3:0] sw,
    output reg[6:0] seg,
    output reg[3:0] auto
    );
    always_comb begin//根据输入选择对应的输出，具体的真值由真值表得到
        case (sw)
            4'h0: seg = 7'b1111110;
            4'h1: seg = 7'b0110000;
            4'h2: seg = 7'b1101101;
            4'h3: seg = 7'b1111001;
            4'h4: seg = 7'b0110011;
            4'h5: seg = 7'b1011011;
            4'h6: seg = 7'b1011111;
            4'h7: seg = 7'b1110000;
            4'h8: seg = 7'b1111111;
            4'h9: seg = 7'b1110011;
            4'hA: seg = 7'b1110111;
            4'hB: seg = 7'b0011111;
            4'hC: seg = 7'b1001110;
            4'hD: seg = 7'b0111101;
            4'hE: seg = 7'b1001111;
            4'hF: seg = 7'b1000111;
            default: seg = 7'b0; 
        endcase
    end
    always_comb begin//根据输入选择对应的输出，只需将16进制翻译成二进制即可
        case (sw)
            4'h0: auto = 4'b0000;
            4'h1: auto = 4'b0001;
            4'h2: auto = 4'b0010;
            4'h3: auto = 4'b0011;
            4'h4: auto = 4'b0100;
            4'h5: auto = 4'b0101;
            4'h6: auto = 4'b0110;
            4'h7: auto = 4'b0111;
            4'h8: auto = 4'b1000;
            4'h9: auto = 4'b1001;
            4'hA: auto = 4'b1010;
            4'hB: auto = 4'b1011;
            4'hC: auto = 4'b1100;
            4'hD: auto = 4'b1101;
            4'hE: auto = 4'b1110;
            4'hF: auto = 4'b1111;
            default: auto = 4'b0;
        endcase
     end
endmodule
